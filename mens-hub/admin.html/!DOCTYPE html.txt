<script>
  // Simple in-browser product list (no backend)
  let products = [];
  let editIndex = null;

  function renderProducts() {
    const tbody = document.getElementById('productList');
    tbody.innerHTML = '';

    products.forEach((p, index) => {
      tbody.innerHTML += `
        <tr>
          <td><img src="${p.image}" class="img-preview" /></td>
          <td>${p.name}</td>
          <td>${p.price} ৳</td>
          <td>
            <button onclick="startEdit(${index})">Edit</button>
            <button style="background:red" onclick="deleteProduct(${index})">Delete</button>
          </td>
        </tr>
      `;
    });
  }

  function saveProduct() {
    const name = document.getElementById('pname').value.trim();
    const price = document.getElementById('pprice').value.trim();
    const image = document.getElementById('pimage').value.trim();

    if (!name || !price || !image) {
      alert('Fill all fields');
      return;
    }

    if (editIndex === null) {
      // create
      products.push({ name, price, image });
    } else {
      // update
      products[editIndex] = { name, price, image };
      editIndex = null;
    }

    document.getElementById('pname').value = '';
    document.getElementById('pprice').value = '';
    document.getElementById('pimage').value = '';

    renderProducts();
  }

  function startEdit(index) {
    const p = products[index];
    document.getElementById('pname').value = p.name;
    document.getElementById('pprice').value = p.price;
    document.getElementById('pimage').value = p.image;
    editIndex = index;
  }

  function deleteProduct(index) {
    if (confirm('Delete this product?')) {
      products.splice(index, 1);
      renderProducts();
    }
  }

  // Test data
  products = [
    { name: 'Demo T-Shirt', price: 800, image: 'https://via.placeholder.com/300x200?text=Demo+T-Shirt' },
    { name: 'Demo Hoodie', price: 1500, image: 'https://via.placeholder.com/300x200?text=Demo+Hoodie' }
  ];
  renderProducts();
</script>
